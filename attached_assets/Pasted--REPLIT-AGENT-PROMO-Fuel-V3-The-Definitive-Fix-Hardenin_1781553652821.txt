```typescript
// ── Global Auth Middleware ────────────────────────────────────────────────
const API_SECRET = process.env.API_SECRET ?? "";

app.use((req: Request, res: Response, next: NextFunction) => {
  // Public paths — no auth required
  const PUBLIC = ["/health", "/api/health"];
  if (PUBLIC.includes(req.path)) return next();

  // TWA paths — authenticated by Telegram initData in Section C5
  if (req.path.startsWith("/api/twa/")) return next();

  // All other paths require Bearer token
  const authHeader = req.headers.authorization ?? "";
  const queryToken = (req.query.token as string) ?? "";
  const token = authHeader.startsWith("Bearer ")
    ? authHeader.slice(7)
    : queryToken;

  if (!API_SECRET || token !== API_SECRET) {
    return res.status(401).json({ error: "Unauthorized" });
  }
  next();
});
```

### C5. API Server — Telegram initData HMAC validation middleware

In **`artifacts/api-server/src/app.ts`**, add this **after** the global auth middleware:

```typescript
import crypto from "crypto";

function validateTelegramInitData(initData: string): number | null {
  const BOT_TOKEN = process.env.TELEGRAM_TOKEN ?? "";
  if (!initData || !BOT_TOKEN) return null;

  const params = new URLSearchParams(initData);
  const hash = params.get("hash");
  if (!hash) return null;

  params.delete("hash");
  const dataCheckString = [...params.entries()]
    .sort(([a], [b]) => a.localeCompare(b))
    .map(([k, v]) => `${k}=${v}`)
    .join("\n");

  const secretKey = crypto
    .createHmac("sha256", "WebAppData")
    .update(BOT_TOKEN)
    .digest();
  const expected = crypto
    .createHmac("sha256", secretKey)
    .update(dataCheckString)
    .digest("hex");

  if (expected !== hash) return null;

  try {
    const user = JSON.parse(params.get("user") ?? "{}");
    return typeof user.id === "number" ? user.id : null;
  } catch {
    return null;
  }
}

// TWA middleware — admin-only Telegram WebApp routes
app.use("/api/twa", (req: Request, res: Response, next: NextFunction) => {
  const initData = (req.headers["x-telegram-init-data"] as string) ?? "";
  const userId = validateTelegramInitData(initData);
  const ADMIN_ID = parseInt(process.env.ADMIN_TELEGRAM_ID ?? "0", 10);

  if (!userId || userId !== ADMIN_ID) {
    return res.status(403).json({ error: "Forbidden: admin only" });
  }
  (req as any).telegramUserId = userId;
  next();
});
```

```typescript
import { Router } from "express";
import multer from "multer";
import { getDb } from "../db";

const router = Router();
const upload = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: 10 * 1024 * 1024 }, // 10MB
});

router.post("/", upload.single("file"), async (req, res) => {
  if (!req.file) return res.status(400).json({ error: "No file uploaded" });

  const filename = req.file.originalname.toLowerCase();
  const allowed = [".html", ".csv", ".tsv", ".json", ".jsonl"];
  const ext = "." + filename.split(".").pop();
  if (!allowed.includes(ext)) {
    return res.status(400).json({ error: `Unsupported format. Allowed: ${allowed.join(", ")}` });
  }

  const content = req.file.buffer.toString("utf-8");
  const key = `upload_${Date.now()}`;
  const entries: any[] = [];

  try {
    if (ext === ".csv" || ext === ".tsv") {
      const delimiter = ext === ".tsv" ? "\t" : ",";
      const lines = content.split(/\r?\n/).filter(Boolean);
      if (lines.length < 2) return res.status(400).json({ error: "File has no data rows" });
      const headers = lines[0].split(delimiter).map(h => h.trim().replace(/^"|"$/g, ""));
      for (const line of lines.slice(1)) {
        const vals = line.split(delimiter);
        const row: Record<string, string> = {};
        headers.forEach((h, i) => {
          const v = (vals[i] ?? "").trim().replace(/^"|"$/g, "");
          if (v) row[h] = v;
        });
        if (Object.keys(row).length > 0) entries.push(row);
      }
    } else if (ext === ".json") {
      const data = JSON.parse(content);
      if (Array.isArray(data)) entries.push(...data);
      else entries.push(data);
    } else if (ext === ".jsonl") {
      for (const line of content.split(/\r?\n/).filter(Boolean)) {
        try { entries.push(JSON.parse(line)); } catch { /* skip malformed lines */ }
      }
    } else if (ext === ".html") {
      const phoneRe = /(?:\+7|8|7)[\s\-]?\(?\d{3}\)?[\s\-]?\d{3}[\s\-]?\d{2}[\s\-]?\d{2}/g;
      const phones = [...new Set(content.match(phoneRe) ?? [])];
      for (const p of phones.slice(0, 500)) {
        entries.push({ phone: p.replace(/[\s\-\(\)]/g, "") });
      }
    }
  } catch (e: any) {
    return res.status(400).json({ error: `Parse error: ${e.message}` });
  }

  const db = getDb();
  db.prepare(`
    INSERT OR REPLACE INTO uploads (key, filename, entries_json, uploaded_at)
    VALUES (?, ?, ?, datetime('now'))
  `).run(key, req.file.originalname, JSON.stringify(entries));

  res.json({ key, filename: req.file.originalname, count: entries.length });
});

router.get("/", (req, res) => {
  const db = getDb();
  const rows = db.prepare(
    "SELECT key, filename, uploaded_at, imported_count, json_array_length(entries_json) as count FROM uploads ORDER BY rowid DESC"
  ).all();
  res.json(rows);
});

export default router;
```

```typescript
import { useState, useRef } from "react";

const ALLOWED_EXT = [".html", ".csv", ".tsv", ".json", ".jsonl"];

export default function UploadPage() {
  const [file, setFile] = useState<File | null>(null);
  const [status, setStatus] = useState<"idle"|"uploading"|"done"|"error">("idle");
  const [result, setResult] = useState<{ key: string; count: number; filename: string } | null>(null);
  const [errorMsg, setErrorMsg] = useState("");
  const inputRef = useRef<HTMLInputElement>(null);

  const apiBase = import.meta.env.VITE_API_URL || "";
  const initData = window.Telegram?.WebApp?.initData ?? "";

  function onFileChange(e: React.ChangeEvent<HTMLInputElement>) {
    const f = e.target.files?.[0];
    if (!f) return;
    const ext = "." + f.name.split(".").pop()?.toLowerCase();
    if (!ALLOWED_EXT.includes(ext)) {
      alert(`Формат не поддерживается. Используйте: ${ALLOWED_EXT.join(", ")}`);
      return;
    }
    setFile(f);
    setStatus("idle");
    setResult(null);
  }

  async function handleUpload() {
    if (!file) return;
    setStatus("uploading");
    setErrorMsg("");
    try {
      const fd = new FormData();
      fd.append("file", file);
      const res = await fetch(`${apiBase}/api/twa/upload`, {
        method: "POST",
        headers: { "X-Telegram-Init-Data": initData },
        body: fd,
      });
      if (!res.ok) {
        const txt = await res.text();
        throw new Error(txt);
      }
      const data = await res.json();
      setResult(data);
      setStatus("done");
    } catch (e: any) {
      setErrorMsg(e.message);
      setStatus("error");
    }
  }

  return (
    <div className="page upload-page">
      <h2>📁 Загрузка аудитории</h2>
      <p className="hint">Поддерживаемые форматы: {ALLOWED_EXT.join(", ")}</p>
      <p className="hint">Максимальный размер: 10 МБ</p>

      <div className="upload-zone" onClick={() => inputRef.current?.click()}>
        <input
          ref={inputRef}
          type="file"
          accept={ALLOWED_EXT.join(",")}
          style={{ display: "none" }}
          onChange={onFileChange}
        />
        {file
          ? <p>📄 <strong>{file.name}</strong> ({(file.size / 1024).toFixed(1)} KB)</p>
          : <p>👆 Нажми чтобы выбрать файл</p>
        }
      </div>

      {file && status === "idle" && (
        <button className="btn-primary" onClick={handleUpload}>⬆️ Загрузить</button>
      )}
      {status === "uploading" && <p>⏳ Загружаю файл...</p>}
      {status === "error" && <p className="error-text">❌ {errorMsg}</p>}
      {status === "done" && result && (
        <div className="upload-result">
          <p>✅ Успешно загружено: <strong>{result.count}</strong> записей</p>
          <p>🔑 Ключ: <code>{result.key}</code></p>
          <p>📄 Файл: {result.filename}</p>
        </div>
      )}
    </div>
  );
}
```

```python
import os
import json

_db_lock = threading.Lock()

def db_load() -> dict:
    try:
        with open(DB_FILE, "r", encoding="utf-8") as f:
            return json.load(f)
    except (FileNotFoundError, json.JSONDecodeError):
        return {}

def db_save(data: dict) -> None:
    """Atomic write via temp file to prevent corruption."""
    tmp = DB_FILE + ".tmp"
    with open(tmp, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
    os.replace(tmp, DB_FILE)

def db_get(key: str, default=None):
    with _db_lock:
        return db_load().get(key, default)

def db_set(key: str, value) -> None:
    with _db_lock:
        data = db_load()
        data[key] = value
        db_save(data)
```

### I4. `campaign_sender.py` — Spintax resolver

Add this function at the top of `campaign_sender.py`:

```python
import re
import random

def resolve_spintax(text: str) -> str:
    """
    Resolves {option1|option2|option3} spintax patterns.
    Handles nested spintax up to 10 levels deep.
    Example: {Привет|Здравствуй}, {name}! → "Привет, {name}!" or "Здравствуй, {name}!"
    """
    for _ in range(10):
        new_text = re.sub(
            r'\{([^{}]+)\}',
            lambda m: random.choice(m.group(1).split("|")),
            text
        )
        if new_text == text:
            break
        text = new_text
    return text
```

```python
async def human_delay(sent_count: int) -> None:
    """
    Mimics human typing/reading rhythm to reduce ban risk.
    - Base: 3–8s per message
    - Every 10 sends: extra 15–40s break
    - Every 50 sends: longer 60–180s break
    - Every 100 sends: rotate account
    """

    extra = 0.0

    if sent_count > 0 and sent_count % 100 == 0:
        extra = random.uniform(120.0, 300.0)
        logger.info(f"[Sender] Major break after {sent_count} sends ({extra:.0f}s)...")
    elif sent_count > 0 and sent_count % 50 == 0:
        extra = random.uniform(60.0, 180.0)
        logger.info(f"[Sender] Long break after {sent_count} sends ({extra:.0f}s)...")
    elif sent_count > 0 and sent_count % 10 == 0:
        extra = random.uniform(15.0, 40.0)
        logger.info(f"[Sender] Short break after {sent_count} sends ({extra:.0f}s)...")

    total = base + extra
```
