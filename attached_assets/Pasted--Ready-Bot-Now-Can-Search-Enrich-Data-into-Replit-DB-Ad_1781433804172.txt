async def enrich_with_telegram(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Enrich all entries that have phone numbers with Telegram info"""
    if not context.args or context.args[0] != "start":
        await update.message.reply_text("Использование: /enrich start\n(Это может занять время)")
        return

    await update.message.reply_text("🚀 Начинаю обогащение БД данными из Telegram...")

    count = 0
    for key in list(db.keys()):
        if not (key.startswith("upload_") or key.startswith("report_")):
            continue
        
        data = db[key]
        entries = data.get("entries", [])
        updated = False
        
        for entry in entries:
            phone = entry.get("phone") or entry.get("Phone") or entry.get("телефон")
            if phone and isinstance(phone, str) and len(phone) > 7:
                try:
                    # Try to get Telegram info
                    client = await init_telethon()  # from previous Telethon code
                    entity = await client.get_entity(phone)
                    entry["telegram"] = {
                        "username": entity.username,
                        "id": entity.id,
                        "name": f"{entity.first_name or ''} {entity.last_name or ''}".strip()
                    }
                    updated = True
                    count += 1
                except:
                    continue  # skip if not found or error
        
        if updated:
            db[key] = data

    await update.message.reply_text(f"✅ Обогащение завершено! Обработано {count} номеров.")
