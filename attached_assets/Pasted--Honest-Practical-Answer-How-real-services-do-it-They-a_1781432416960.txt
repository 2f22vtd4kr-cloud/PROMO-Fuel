from telethon import TelegramClient
from telethon.errors import SessionPasswordNeededError
import asyncio

# Global client (initialize once)
tg_client = None

async def init_telethon():
    global tg_client
    if tg_client is None:
        api_id = os.getenv("API_ID")
        api_hash = os.getenv("API_HASH")
        phone = os.getenv("PHONE")
        tg_client = TelegramClient('session', api_id, api_hash)
        await tg_client.connect()
        if not await tg_client.is_user_authorized():
            await tg_client.send_code_request(phone)
            code = input("Enter the code you received in Telegram: ")  # First time only
            try:
                await tg_client.sign_in(phone, code)
            except SessionPasswordNeededError:
                password = input("Enter 2FA password: ")
                await tg_client.sign_in(password=password)
    return tg_client

# Command: /tgphone +7916xxxxxxx
async def tg_phone_lookup(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not context.args:
        await update.message.reply_text("Использование: /tgphone +79161234567")
        return

    phone = context.args[0].strip()
    if not phone.startswith('+'):
        phone = '+' + phone

    await update.message.reply_text(f"🔍 Проверяю номер {phone} в Telegram...")

    try:
        client = await init_telethon()
        entity = await client.get_entity(phone)
        
        result = (
            f"✅ Найден Telegram аккаунт:\n\n"
            f"👤 Имя: {entity.first_name or ''} {entity.last_name or ''}\n"
            f"🔗 Username: @{entity.username if entity.username else 'Нет'}\n"
            f"🆔 ID: `{entity.id}`\n"
            f"📱 Номер: {phone}\n"
            f"🤖 Бот: {'Да' if entity.bot else 'Нет'}\n"
            f"⭐ Премиум: {'Да' if getattr(entity, 'premium', False) else 'Нет'}"
        )
        await update.message.reply_text(result, parse_mode='Markdown')
        
    except Exception as e:
        await update.message.reply_text(f"❌ Не найдено или скрыто:\n{str(e)[:200]}")

# Add to main():
app.add_handler(CommandHandler("tgphone", tg_phone_lookup))
