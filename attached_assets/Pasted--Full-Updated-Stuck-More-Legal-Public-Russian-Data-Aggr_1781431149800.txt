async def inn_lookup(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not context.args:
        await update.message.reply_text("Использование: /inn 7707083893")
        return
    
    query = context.args[0].strip()
    await update.message.reply_text(f"🔍 Ищу по ИНН/ОГРН: {query}...")

    # Free public sources (priority order)
    sources = [
        f"https://htmlweb.ru/api/service/org?inn={query}",           # 20 free/day, no key
        f"https://htmlweb.ru/api/service/org?ogrn={query}",          # fallback OGRN
        # Add more after user registers free tier (optional key support later)
    ]

    found = False
    result_text = ""
    for url in sources:
        try:
            resp = requests.get(url, timeout=15)
            if resp.status_code == 200:
                data = resp.json()
                
                name = data.get('name') or data.get('full_name', 'N/A')
                inn = data.get('inn', query)
                ogrn = data.get('ogrn', 'N/A')
                address = data.get('address', 'N/A')
                status = data.get('status', 'N/A')
                okved = data.get('okved', 'N/A')
                
                result_text = (
                    f"✅ Найдено (публичные данные):\n"
                    f"Название: {name}\n"
                    f"ИНН: {inn}\n"
                    f"ОГРН: {ogrn}\n"
                    f"Адрес: {address}\n"
                    f"Статус: {status}\n"
                    f"ОКВЭД: {okved}\n\n"
                    f"Лимит: {data.get('limit', 'N/A')} запросов осталось сегодня."
                )
                await update.message.reply_text(result_text)
                found = True
                break
        except Exception:
            continue

    if not found:
        await update.message.reply_text(
            "🔍 Данные не найдены в бесплатном лимите или источник недоступен.\n\n"
            "Рекомендации (бесплатные тиры):\n"
            "• Зарегистрируйся на Checko.ru или Ofdata.ru → Лайт тариф (бесплатно)\n"
            "• Rusprofile.ru, Контур.Фокус (демо), egrul.nalog.ru (официальный поиск)\n"
            "• htmlweb.ru даёт 20 запросов в сутки"
        )
    await send_promo(update)
