import os
import logging
import requests
from dotenv import load_dotenv
from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes, CallbackQueryHandler

load_dotenv()
TOKEN = os.getenv("TELEGRAM_TOKEN")

logging.basicConfig(level=logging.INFO)

# ==================== PROMO ====================
async def send_promo(update: Update):
    promo_text = (
        "🔥 Нравится бот? Попробуй мой открытый проект **fuel-tickets-ru** — "
        "удобный инструмент для экономии на топливе и билетах.\n\n"
        "GitHub: https://github.com/2f22vtd4kr-cloud/fuel-tickets-ru\n"
        "Поставь ⭐ если полезно!"
    )
    if update.message:
        await update.message.reply_text(promo_text, parse_mode='Markdown')
    elif update.callback_query:
        await update.callback_query.message.reply_text(promo_text, parse_mode='Markdown')

# ==================== COMMANDS ====================
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    welcome = (
        "👋 Привет! Я RUProbeHelper — полезный бот для работы с публичными данными.\n\n"
        "Доступные команды:\n"
        "/inn [ИНН] — поиск организации\n"
        "/ogrn [ОГРН] — поиск по ОГРН\n"
        "/help — список команд\n"
        "/fuel — информация о проекте fuel-tickets-ru\n\n"
        "Только открытые источники. Никаких приватных баз."
    )
    await update.message.reply_text(welcome)
    await send_promo(update)

async def fuel_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await send_promo(update)

async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    help_text = (
        "Бот использует только публичные открытые источники (ЕГРЮЛ-подобные).\n\n"
        "Команды:\n"
        "/inn 7707083893\n"
        "/ogrn 1027739468877\n"
        "/fuel — промо проекта"
    )
    await update.message.reply_text(help_text)

# ==================== PUBLIC COMPANY LOOKUP (Free tier friendly) ====================
async def inn_lookup(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not context.args:
        await update.message.reply_text("Использование: /inn 7707083893")
        return
    
    query = context.args[0]
    await update.message.reply_text(f"🔍 Ищу по ИНН/ОГРН: {query}...")

    # Try public-friendly endpoints (add your preferred free API key if needed)
    urls = [
        f"https://htmlweb.ru/api/service/org?inn={query}",  # example public endpoint
        # Add more if you register for free tiers: checko.ru, ofdata.ru, etc.
    ]

    found = False
    for url in urls:
        try:
            resp = requests.get(url, timeout=10)
            if resp.status_code == 200:
                data = resp.json()
                result = (
                    f"✅ Найдено:\n"
                    f"Название: {data.get('name', 'N/A')}\n"
                    f"ИНН: {data.get('inn', 'N/A')}\n"
                    f"ОГРН: {data.get('ogrn', 'N/A')}\n"
                    f"Адрес: {data.get('address', 'N/A')}\n"
                    f"Статус: {data.get('status', 'N/A')}"
                )
                await update.message.reply_text(result)
                found = True
                break
        except:
            continue

    if not found:
        await update.message.reply_text(
            "Данные не найдены по открытому API или лимит. "
            "Попробуй популярные сервисы: Контур.Фокус, СБИС, Checko.ru"
        )
    await send_promo(update)

# ==================== INLINE MENU ====================
async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    await query.answer()
    
    if query.data == "fuel_promo":
        await send_promo(update)

async def menu(update: Update, context: ContextTypes.DEFAULT_TYPE):
    keyboard = [
        [InlineKeyboardButton("🔎 Поиск по ИНН", callback_data="inn_help")],
        [InlineKeyboardButton("🚀 fuel-tickets-ru", callback_data="fuel_promo")],
    ]
    reply_markup = InlineKeyboardMarkup(keyboard)
    await update.message.reply_text("Выбери действие:", reply_markup=reply_markup)

def main():
    app = Application.builder().token(TOKEN).build()

    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("fuel", fuel_command))
    app.add_handler(CommandHandler("help", help_command))
    app.add_handler(CommandHandler("inn", inn_lookup))
    app.add_handler(CommandHandler("menu", menu))
    
    app.add_handler(CallbackQueryHandler(button_handler))

    # Echo for general messages (optional useful replies)
    async def echo(update: Update, context: ContextTypes.DEFAULT_TYPE):
        if update.message.text and not update.message.text.startswith('/'):
            await update.message.reply_text("Отправь команду, например /inn 7707083893")
            await send_promo(update)
    
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, echo))

    print("🤖 Бот запущен...")
    app.run_polling()

if __name__ == "__main__":
    main()
