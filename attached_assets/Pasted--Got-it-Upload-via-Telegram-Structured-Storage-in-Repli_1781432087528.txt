import re
import csv
import json
from io import StringIO
from replit import db
from datetime import datetime

async def handle_document(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not update.message.document:
        return
    
    doc = update.message.document
    filename = doc.file_name.lower() if doc.file_name else ""
    
    supported = ('.html', '.csv', '.tsv', '.jsonl', '.json')
    if not any(filename.endswith(ext) for ext in supported):
        await update.message.reply_text("❌ Підтримуються файли: **.html**, **.csv**, **.tsv**, **.jsonl**, **.json**")
        return

    await update.message.reply_text(f"📤 Отримую файл: `{filename}`\nОбробка...")

    try:
        file = await context.bot.get_file(doc.file_id)
        content = (await file.download_as_bytearray()).decode('utf-8', errors='ignore')
        
        extracted = {
            "filename": filename,
            "uploaded_at": datetime.now().isoformat(),
            "entries": []
        }

        # HTML (OSINT reports)
        if filename.endswith('.html'):
            # Extract names, phones, addresses, vehicles etc.
            names = re.findall(r'([А-ЯЁ][А-ЯЁа-яё]{1,}\s+[А-ЯЁ][А-ЯЁа-яё]{1,}(?:\s+[А-ЯЁ][А-ЯЁа-яё]{1,})?)', content)
            phones = re.findall(r'[\+]?[0-9\s\-\(\)]{7,20}', content)
            for name in dict.fromkeys(names):  # remove duplicates preserving order
                extracted["entries"].append({"full_name": name.strip()})
            for phone in phones:
                cleaned = re.sub(r'\s+', '', phone)
                if len(cleaned) > 7:
                    extracted["entries"].append({"phone": cleaned})

        # CSV / TSV
        elif filename.endswith(('.csv', '.tsv')):
            delimiter = '\t' if filename.endswith('.tsv') else ','
            reader = csv.DictReader(StringIO(content), delimiter=delimiter)
            for row in reader:
                clean_row = {k.strip(): str(v).strip() for k, v in row.items() if v and str(v).strip()}
                if clean_row:
                    extracted["entries"].append(clean_row)

        # JSONL / JSON
        elif filename.endswith(('.jsonl', '.json')):
            if filename.endswith('.jsonl'):
                lines = content.strip().split('\n')
                for line in lines:
                    if line.strip():
                        try:
                            data = json.loads(line)
                            extracted["entries"].append(data if isinstance(data, dict) else {"data": data})
                        except:
                            continue
            else:  # single JSON
                try:
                    data = json.loads(content)
                    if isinstance(data, list):
                        extracted["entries"] = data
                    elif isinstance(data, dict):
                        extracted["entries"] = [data]
                except:
                    pass

        # Save to Replit DB
        key = f"upload_{int(__import__('time').time())}"
        db[key] = extracted

        count = len(extracted["entries"])
        await update.message.reply_text(
            f"✅ **Файл завантажено та оброблено!**\n\n"
            f"Записів збережено: **{count}**\n"
            f"Ключ у базі: `{key}`\n\n"
            f"Переглянути:\n"
            f"`/getdata {key}`\n"
            f"`/listdata` — список всіх завантажень",
            parse_mode='Markdown'
        )

    except Exception as e:
        await update.message.reply_text(f"❌ Помилка при обробці файлу:\n{str(e)[:300]}")
How to Use
	1	Add/replace this function in main.py
	2	Ensure the handler exists: app.add_handler(MessageHandler(filters.Document.ALL, handle_document))
	3	
	4	Restart the bot.
